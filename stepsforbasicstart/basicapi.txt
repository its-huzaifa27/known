>and same as errorHandler middleware we also import the jwt and asyncHandler from their respective lib

>const validateToken = asyncHandler(async (req,res,next) => {
    //first create a token variable 
    let token;
    //after that whenever the user sends an req the access token is passedthrough the header section of req with auth field either we pass in the auth section or we can create a authorization in the header in the header we create authorization and make it equal to Bearer 
    let authHeader = req.headers.Authorization || req.headers.Authorization
    //check if authheader is present and also check if it starts with Bearer  
    if(authHeader && authHeader.startsWith("Bearer")){
        //if it is starts with Bearer then take the token and make the authHeader aplit the array with " " so the first index means [0] will be "Bearer" and 2nd index [1] will be "the acces token "
        token = authHeader.split(" ")[1];//that means the token = the access token generated by login
        //next verify it so to verify it we do 
        jwt.verify(token,process.env.ACCESS_TOKEN,(err,decoded) => {
            if (err) {
                res.status(401)
                throw new error("user is not authorized")
            }
            req.user = decoded.user;
        })
        

    }
})
